CREATE TABLE `gallery`.`users` ( `user_id` INT NOT NULL AUTO_INCREMENT , `email` VARCHAR(100) NOT NULL , `username` VARCHAR(50) NOT NULL , `password` VARCHAR(250) NOT NULL , `created_at` DATETIME on update CURRENT_TIMESTAMP NOT NULL , `change_bg` INT NOT NULL , PRIMARY KEY (`user_id`), UNIQUE `email` (`email`)) ENGINE = InnoDB;
